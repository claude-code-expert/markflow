    ```typescript
    import type { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
    import { authMiddleware } from '../middleware/auth.js';
    import { unauthorized } from '../utils/errors.js';
    import { logger } from '../utils/logger.js';

    export async function uploadTokenRoutes(app: FastifyInstance) {
      app.addHook('preHandler', authMiddleware);

      // GET /api/v1/upload-token
      // 인증된 사용자에게 R2 Worker 업로드용 토큰을 반환
      // Worker의 API_SECRET과 동일한 값을 환경변수에서 읽어 반환
      app.get('/', async (request: FastifyRequest, reply: FastifyReply) => {
        if (!request.currentUser) {
          throw unauthorized('Not authenticated');
        }

        const uploadSecret = process.env.R2_UPLOAD_SECRET;
        if (!uploadSecret) {
          // R2_UPLOAD_SECRET 미설정 시 토큰 없이 업로드 가능 (D-05 하위 호환)
          logger.warn('R2_UPLOAD_SECRET not configured — upload token endpoint returns empty token');
          return reply.status(200).send({ token: null });
        }

        return reply.status(200).send({ token: uploadSecret });
      });
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> R2 Worker | 브라우저에서 직접 Worker로 파일 업로드 요청. 인증 없이 접근 가능했던 경계. |
| Browser -> API Server | API 서버를 통해 업로드 토큰 요청. JWT 인증 경계. |
| API Server -> Worker (shared secret) | API_SECRET / R2_UPLOAD_SECRET 동일 값으로 신뢰 관계 성립. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | Worker CORS | mitigate | ALLOWED_ORIGINS에서 wildcard `*` 제거, fail-closed (빈 배열 시 모든 origin 거부) |
| T-01-02 | Elevation | Worker upload | mitigate | API_SECRET 설정 시 Bearer token 검증, 불일치 시 403 반환 |
| T-01-03 | Information Disclosure | Worker timing attack | accept | Workers 런타임 고정 실행 시간으로 실제 공격 벡터 극히 작음. MVP 규모에서 수용 가능. |
| T-01-04 | Spoofing | upload-token endpoint | mitigate | authMiddleware로 JWT 인증 필수, 미인증 시 401 |
| T-01-05 | Information Disclosure | R2_UPLOAD_SECRET 노출 | mitigate | 환경변수로만 관리, 코드에 하드코딩 금지, API 응답으로만 인증된 사용자에게 전달 |
</threat_model>
